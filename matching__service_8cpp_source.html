42std::vector<core::HashMatch> MatchingService::BuildHashMatches(
43 const std::vector<core::Fingerprint>& fragment_fps,
44 const std::vector<HashLookupResult>& lookup_results) {
45 // Построить индекс: hash → список anchor_frame во фрагменте.
46 // Один хэш может встречаться в нескольких позициях фрагмента.
47 std::unordered_map<uint32_t, std::vector<std::size_t>> hash_to_fragment_frames;
48 for (const auto& [hash_, anchor_frame_] : fragment_fps) {
49 hash_to_fragment_frames[hash_].push_back(anchor_frame_);
50 }
51
52 // Для каждого совпадения из БД — найти все позиции этого хэша во фрагменте
53 // и создать HashMatch для каждой пары (трек-позиция, фрагмент-позиция).
54 std::vector<core::HashMatch> matches;
55 for (const auto& [hash_, track_id_, track_anchor_frame_] : lookup_results) {
56 const auto it = hash_to_fragment_frames.find(hash_);
57 if (it == hash_to_fragment_frames.end()) {
58 continue;
59 }
60 for (const std::size_t fragment_frame : it->second) {
61 matches.push_back(core::HashMatch{
62 .track_id_ = track_id_,
63 .track_anchor_frame_ = track_anchor_frame_,
64 .fragment_anchor_frame_ = fragment_frame,
65 });
66 }
67 }
