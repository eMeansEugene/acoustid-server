26 IndexingResult IndexingService::DoIndex(const audio::AudioData& audio,
27 const TrackMetadata& metadata) {
28 // Если duration не задан явно — берём из декодированного аудио.
29 TrackMetadata effective = metadata;
30 if (effective.duration_sec_ <= 0.0F) {
31 effective.duration_sec_ = audio.duration_sec_;
32 }
33
34 const core::FingerprintResult result = engine_.Process(audio.samples_);
35
36 const std::size_t track_id = repository_.AddTrackWithFingerprints(effective, result.fingerprints);
37
38 return IndexingResult{track_id, result.fingerprints.size()};
39 }
