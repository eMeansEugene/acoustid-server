66void SQLiteRepository::CreateSchema() {
67 Execute(R"(
68 CREATE TABLE IF NOT EXISTS tracks (
69 id INTEGER PRIMARY KEY AUTOINCREMENT,
70 title TEXT NOT NULL,
71 artist TEXT NOT NULL,
72 duration REAL NOT NULL,
73 indexed_at INTEGER NOT NULL
74 );
75 )");
76
77 Execute(R"(
78 CREATE TABLE IF NOT EXISTS fingerprints (
79 id INTEGER PRIMARY KEY AUTOINCREMENT,
80 track_id INTEGER NOT NULL,
81 hash INTEGER NOT NULL,
82 time_offset INTEGER NOT NULL,
83 FOREIGN KEY (track_id) REFERENCES tracks(id) ON DELETE CASCADE
84 );
85 )");
86
87 Execute("CREATE INDEX IF NOT EXISTS idx_fingerprints_hash ON fingerprints(hash);");
88 Execute("CREATE INDEX IF NOT EXISTS idx_fingerprints_track_id ON fingerprints(track_id);");
89
90 // Включаем поддержку foreign keys (для каскадного удаления).
91 Execute("PRAGMA foreign_keys=ON;");
92}

106std::size_t SQLiteRepository::AddTrackWithFingerprints(const domain::TrackMetadata& metadata,
107 const std::vector<core::Fingerprint>& fingerprints) {
108 std::lock_guard lock(mutex_);
109
110 Execute("BEGIN TRANSACTION;");
111
112 try {
113 // 1. Вставить трек.
114 sqlite3_stmt* raw_stmt = nullptr;
115 int rc = sqlite3_prepare_v2(
116 db_, "INSERT INTO tracks (title, artist, duration, indexed_at) VALUES (?, ?, ?, ?);", -1, &raw_stmt,
117 nullptr);
118 if (rc != SQLITE_OK) {
119 throw std::runtime_error("Failed to prepare INSERT tracks: " + std::string(sqlite3_errmsg(db_)));
120 }
121 StmtGuard track_stmt(raw_stmt);
122
123 sqlite3_bind_text(track_stmt.Get(), 1, metadata.title_.c_str(), -1, SQLITE_TRANSIENT);
124 sqlite3_bind_text(track_stmt.Get(), 2, metadata.artist_.c_str(), -1, SQLITE_TRANSIENT);
125 sqlite3_bind_double(track_stmt.Get(), 3, static_cast<double>(metadata.duration_sec_));
126 sqlite3_bind_int64(track_stmt.Get(), 4, NowUnixSeconds());
127
128 rc = sqlite3_step(track_stmt.Get());
129 if (rc != SQLITE_DONE) {
130 throw std::runtime_error("Failed to insert track: " + std::string(sqlite3_errmsg(db_)));
131 }
132
133 const auto track_id = static_cast<std::size_t>(sqlite3_last_insert_rowid(db_));
134
135 // 2. Вставить fingerprints батчом.
136 rc = sqlite3_prepare_v2(
137 db_, "INSERT INTO fingerprints (track_id, hash, time_offset) VALUES (?, ?, ?);", -1, &raw_stmt, nullptr);
138 if (rc != SQLITE_OK) {
139 throw std::runtime_error("Failed to prepare INSERT fingerprints: " + std::string(sqlite3_errmsg(db_)));
140 }
141 StmtGuard fp_stmt(raw_stmt);
142
143 for (const core::Fingerprint& fp : fingerprints) {
144 sqlite3_reset(fp_stmt.Get());
145 sqlite3_bind_int64(fp_stmt.Get(), 1, static_cast<sqlite3_int64>(track_id));
146 sqlite3_bind_int64(fp_stmt.Get(), 2, static_cast<sqlite3_int64>(fp.hash_));
147 sqlite3_bind_int64(fp_stmt.Get(), 3, static_cast<sqlite3_int64>(fp.anchor_frame_));
148
149 rc = sqlite3_step(fp_stmt.Get());
150 if (rc != SQLITE_DONE) {
151 throw std::runtime_error("Failed to insert fingerprint: " + std::string(sqlite3_errmsg(db_)));
152 }
153 }
154
155 Execute("COMMIT;");
156 return track_id;
157
158 } catch (...) {
159 Execute("ROLLBACK;");
160 throw;
161 }
162}

166std::vector<domain::HashLookupResult> SQLiteRepository::FindMatches(const std::vector<uint32_t>& hashes) {
167 std::lock_guard lock(mutex_);
168
169 std::vector<domain::HashLookupResult> results;
170
171 sqlite3_stmt* raw_stmt = nullptr;
172 const int rc = sqlite3_prepare_v2(
173 db_, "SELECT hash, track_id, time_offset FROM fingerprints WHERE hash = ?;", -1, &raw_stmt, nullptr);
174 if (rc != SQLITE_OK) {
175 throw std::runtime_error("Failed to prepare SELECT fingerprints: " + std::string(sqlite3_errmsg(db_)));
176 }
177 StmtGuard stmt(raw_stmt);
178
179 for (const uint32_t hash : hashes) {
180 sqlite3_reset(stmt.Get());
181 sqlite3_bind_int64(stmt.Get(), 1, static_cast<sqlite3_int64>(hash));
182
183 while (sqlite3_step(stmt.Get()) == SQLITE_ROW) {
184 results.push_back(domain::HashLookupResult{
185 .hash_ = static_cast<uint32_t>(sqlite3_column_int64(stmt.Get(), 0)),
186 .track_id_ = static_cast<std::size_t>(sqlite3_column_int64(stmt.Get(), 1)),
187 .track_anchor_frame_ = static_cast<std::size_t>(sqlite3_column_int64(stmt.Get(), 2)),
188 });
189 }
190 }
191
192 return results;
193}
