64crow::response HttpServer::HandleMatch(const crow::request& req) {
65 crow::multipart::message msg(req);
66
67 // Ищем part с файлом.
68 const crow::multipart::part* file_part = nullptr;
69 for (const auto& part : msg.parts) {
70 auto it = part.headers.find("Content-Disposition");
71 if (it != part.headers.end() && it->second.params.count("filename")) {
72 file_part = &part;
73 break;
74 }
75 }
76
77 if (!file_part || file_part->body.empty()) {
78 json err;
79 err["error"] = "No audio file provided";
80 return crow::response(400, err.dump());
81 }
82
83 if (file_part->body.size() > config_.max_upload_bytes) {
84 json err;
85 err["error"] = "File exceeds maximum size";
86 return crow::response(413, err.dump());
87 }
88
89 // Создать задачу.
90 const std::string task_id = GenerateTaskId();
91 std::vector<uint8_t> bytes(file_part->body.begin(), file_part->body.end());
92
93 registry_.Register(task_id);
94 queue_.Push(Task{task_id, std::move(bytes)});
95
96 std::cout << "[server] POST /match -> task_id=" << task_id << "\n";
97
98 json body;
99 body["task_id"] = task_id;
100 auto resp = crow::response(202, body.dump());
101 resp.set_header("Content-Type", "application/json");
102 return resp;
103}

107crow::response HttpServer::HandleGetTask(const std::string& task_id) {
108 const auto state = registry_.Get(task_id);
109 if (!state.has_value()) {
110 json err;
111 err["error"] = "Task not found";
112 return crow::response(404, err.dump());
113 }
114
115 json body;
116 body["task_id"] = task_id;
117
118 switch (state->status) {
119 case TaskStatus::PENDING:
120 body["status"] = "pending";
121 break;
122 case TaskStatus::PROCESSING:
123 body["status"] = "processing";
124 break;
125 case TaskStatus::DONE:
126 body["status"] = "done";
127 if (state->output.has_value() && state->output->match_result.has_value()) {
128 const auto& mr = *state->output->match_result;
129 // Для получения метаданных трека используем GetAllTracks
130 // и ищем по id. В будущем можно добавить GetTrackById.
131 json result;
132 result["track_id"] = mr.track_id_;
133 result["offset_frames"] = mr.offset_frames_;
134 result["confidence"] = mr.confidence_;
135 result["votes"] = mr.votes_;
136
137 const auto tracks = repository_.GetAllTracks();
138 for (const auto& t : tracks) {
139 if (t.id_ == mr.track_id_) {
140 result["track"]["id"] = t.id_;
141 result["track"]["title"] = t.title_;
142 result["track"]["artist"] = t.artist_;
143 result["track"]["duration_sec"] = t.duration_sec_;
144 break;
145 }
146 }
147
148 body["result"] = result;
149 } else {
150 body["result"] = nullptr;
151 }
152 break;
153 case TaskStatus::ERROR:
154 body["status"] = "error";
155 body["error"] = state->error_message;
156 break;
157 }
158
159 auto resp = crow::response(200, body.dump());
160 resp.set_header("Content-Type", "application/json");
161 return resp;
162}

185// --- POST /admin/index ----------------------------------------------------
186
187crow::response HttpServer::HandleAdminIndex(const crow::request& req) const {
188 // Проверить API-ключ.
189 const auto key = req.get_header_value("X-Api-Key");
190 if (key != config_.admin_api_key) {
191 json err;
192 err["error"] = "Invalid or missing API key";
193 return crow::response(401, err.dump());
194 }
195
196 crow::multipart::message msg(req);
197
198 // Найти файл.
199 const crow::multipart::part* file_part = nullptr;
200 std::string title;
201 std::string artist;
202
203 for (const auto& part : msg.parts) {
204 auto it = part.headers.find("Content-Disposition");
205 if (it == part.headers.end()) continue;
206
207 const auto& params = it->second.params;
208 auto name_it = params.find("name");
209 if (name_it == params.end()) continue;
210
211 if (name_it->second == "file" || params.count("filename")) {
212 file_part = &part;
213 } else if (name_it->second == "title") {
214 title = part.body;
215 } else if (name_it->second == "artist") {
216 artist = part.body;
217 }
218 }
219
220 if (!file_part || file_part->body.empty()) {
221 json err;
222 err["error"] = "No audio file provided";
223 return crow::response(400, err.dump());
224 }
225 if (title.empty()) {
226 json err;
227 err["error"] = "Missing required field: title";
228 return crow::response(400, err.dump());
229 }
230 if (artist.empty()) {
231 json err;
232 err["error"] = "Missing required field: artist";
233 return crow::response(400, err.dump());
234 }
235
236 try {
237 std::vector<uint8_t> bytes(file_part->body.begin(), file_part->body.end());
238 const auto result = indexing_.IndexFromBytes(bytes, {title, artist, 0.0F});
239
240 std::cout << "[server] POST /admin/index -> track_id=" << result.track_id
241 << " fingerprints=" << result.fingerprint_count << "\n";
242
243 json body;
244 body["track_id"] = result.track_id;
245 body["fingerprint_count"] = result.fingerprint_count;
246
247 auto resp = crow::response(200, body.dump());
248 resp.set_header("Content-Type", "application/json");
249 return resp;
250
251 } catch (const std::exception& e) {
252 json err;
253 err["error"] = std::string("Indexing failed: ") + e.what();
254 return crow::response(400, err.dump());
255 }
256}

260std::string HttpServer::GenerateTaskId() {
261 static std::mt19937 gen(std::random_device{}());
262 std::uniform_int_distribution<uint64_t> dist;
263 std::ostringstream oss;
264 oss << std::hex << std::setfill('0') << std::setw(12) << (dist(gen) & 0xFFFFFFFFFFFFULL);
265 return oss.str();
266}
