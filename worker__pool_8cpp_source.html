31 void WorkerPool::WorkerLoop() const {

34 if (!task.has_value()) {
35 break; // очередь остановлена
36 }
37
38 const std::string& task_id = task->id;

40
41 try {
42 domain::MatchOutput output = matcher_.Match(task->audio_bytes);
43 registry_.SetDone(task_id, std::move(output));
44 } catch (const std::exception& e) {

46 std::cerr << "[worker] Task " << task_id << " failed: " << e.what() << "\n";
47 } catch (...) {
48 registry_.SetError(task_id, "Unknown error");
49 std::cerr << "[worker] Task " << task_id << " failed: unknown error\n";
