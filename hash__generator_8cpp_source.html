28uint32_t HashGenerator::PackHash(const std::size_t freq_anchor, const std::size_t freq_target,
29 const std::size_t time_delta) {
30 // [31 .. 23] [22 .. 14] [13 .. 0]
31 // freq_anchor freq_target time_delta
32 return static_cast<uint32_t>(((freq_anchor & FREQ_MASK) << (FREQ_BITS + DELTA_BITS)) |
33 ((freq_target & FREQ_MASK) << DELTA_BITS) |
34 (time_delta & DELTA_MASK));
35}

37HashGenerator::UnpackedHash HashGenerator::UnpackHash(const uint32_t hash) {
38 return {
39 .freq_anchor = (hash >> (FREQ_BITS + DELTA_BITS)) & FREQ_MASK,
40 .freq_target = (hash >> DELTA_BITS) & FREQ_MASK,
41 .time_delta = hash & DELTA_MASK,
42 };

45std::vector<Fingerprint> HashGenerator::Generate(const std::vector<Peak>& peaks) const {
46 // Отфильтровать пики с bin_index >= freq_bin_limit_ (верхние частоты).
47 std::vector<const Peak*> valid;
48 valid.reserve(peaks.size());
49 for (const Peak& peak : peaks) {
50 if (peak.bin_index_ < config_.freq_bin_limit_) {
51 valid.push_back(&peak);
52 }
53 }
54
55 std::vector<Fingerprint> fingerprints;
56
57 for (std::size_t i = 0; i < valid.size(); ++i) {
58 const Peak& anchor = *valid[i];
59 std::size_t targets_found = 0;
60
61 // Пики отсортированы по (frame_index, bin_index) на выходе из
62 // PeakExtractor. Цели ищем начиная со следующего пика — тот же
63 // фрейм тоже допустим (две ноты одновременно).
64 for (std::size_t j = i + 1; j < valid.size() && targets_found < config_.max_targets_per_anchor_; ++j) {
65 const Peak& target = *valid[j];
66
67 const std::size_t delta = target.frame_index_ - anchor.frame_index_;
68
69 // За пределами окна — дальше все пики ещё дальше по времени,
70 // можно прервать.
71 if (delta > config_.max_target_offset_) {
72 break;
73 }
74
75 const uint32_t hash = PackHash(anchor.bin_index_, target.bin_index_, delta);
76 fingerprints.push_back(Fingerprint{hash, anchor.frame_index_});
77 ++targets_found;
78 }
79 }
80
81 return fingerprints;
82}
