40std::vector<float> FftEngine::MakeHannWindow(std::size_t size) {
41 std::vector<float> window(size);
42 // periodic Hann: w[n] = 0.5 * (1 - cos(2*pi*n / N)), n in [0, N)
43 // используем N (не N-1) в знаменателе — периодическое окно, стандарт
44 const auto n = static_cast<float>(size);
45 for (std::size_t i = 0; i < size; ++i) {
46 window[i] = 0.5F * (1.0F - std::cos(2.0F * static_cast<float>(M_PI) * static_cast<float>(i) / n));
47 }
48 return window;
49}

51void FftEngine::ProcessFrame(const float* frame_samples, float* out_row) const {
52 const std::size_t frame_size = config_.frame_size_;
53 const std::size_t num_complex_bins = frame_size / 2 + 1; // r2c output size
54
55 std::vector<float> windowed(frame_size);
56 for (std::size_t i = 0; i < frame_size; ++i) {
57 windowed[i] = frame_samples[i] * hann_window_[i];
58 }
59
60 std::vector<std::complex<float>> spectrum(num_complex_bins);
61
62 const pocketfft::shape_t shape{frame_size};
63 const pocketfft::stride_t stride_in{sizeof(float)};
64 const pocketfft::stride_t stride_out{sizeof(std::complex<float>)};
65
66 pocketfft::r2c<float>(shape, stride_in, stride_out, /*axis=*/0, /*forward=*/true, windowed.data(),
67 spectrum.data(), /*fct=*/1.0F);
68
69 // Бин 0 (DC) и бин frame_size/2 (Найквист) отбрасываются: они не несут
70 // информации о гармониках сигнала и не нужны для constellation map.
71 for (std::size_t bin = 0; bin < frame_size / 2; ++bin) {
72 const std::complex<float> c = spectrum[bin + 1];
73 const float power = c.real() * c.real() + c.imag() * c.imag();
74 out_row[bin] = 10.0F * std::log10(power + LOG_POWER_EPSILON);
75 }
76}

78Spectrogram FftEngine::ComputeSpectrogram(const std::vector<float>& samples) const {
79 const std::size_t frame_size = config_.frame_size_;
80 const std::size_t hop_size = config_.hop_size_;
81 const std::size_t num_bins = NumBins();
82
83 if (samples.size() < frame_size) {
84 return Spectrogram(0, num_bins);
85 }
86
87 const std::size_t num_frames = (samples.size() - frame_size) / hop_size + 1;
88 Spectrogram spectrogram(num_frames, num_bins);
89
90 for (std::size_t frame = 0; frame < num_frames; ++frame) {
91 const float* frame_start = samples.data() + frame * hop_size;
92 ProcessFrame(frame_start, &spectrogram.At(frame, 0));
93 }
94
95 return spectrogram;
96}
