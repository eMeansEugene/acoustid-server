44std::optional<MatchResult> VotingEngine::Vote(const std::vector<HashMatch>& matches,
45 const std::size_t total_hashes) const {
46 if (matches.empty() || total_hashes == 0) {
47 return std::nullopt;
48 }
49
50 // Подсчёт голосов по (track_id, Δ).
51 std::unordered_map<VoteKey, std::size_t, VoteKeyHash> votes;
52 for (const auto& [track_id_, track_anchor_frame_, fragment_anchor_frame_] : matches) {
53 const auto delta = static_cast<int64_t>(track_anchor_frame_) -
54 static_cast<int64_t>(fragment_anchor_frame_);
55 votes[{track_id_, delta}]++;
56 }
57
58 // Поиск победителя.
59 VoteKey best_key{};
60 std::size_t best_votes = 0;
61 for (const auto& [key, count] : votes) {
62 if (count > best_votes) {
63 best_votes = count;
64 best_key = key;
65 }
66 }
67
68 const double confidence = static_cast<double>(best_votes) / static_cast<double>(total_hashes);
69
70 if (confidence < config_.min_confidence_) {
71 return std::nullopt;
72 }
73
74 return MatchResult{
75 .track_id_ = best_key.track_id,
76 .offset_frames_ = best_key.delta,
77 .confidence_ = confidence,
78 .votes_ = best_votes,
79 };
80}
